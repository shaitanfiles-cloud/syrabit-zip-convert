```python
# server.py - Async parallel generation
@app.post("/admin/full-subject")
async def generate_full_subject(subject_id: str):
    chapters = await db.chapters.find({"subject_id": subject_id}).to_list(None)
    
    # Parallel: 12 chapters x 5 assets = 60 concurrent
    tasks = []
    for chapter in chapters:
        for asset_type in ["notes", "mcqs", "flashcards", "pyq", "geo-blogs"]:
            tasks.append(gemini_generate_asset(chapter, asset_type))
    
    results = await asyncio.gather(*tasks, return_exceptions=True)
    return {"assets": len([r for r in results if not isinstance(r, Exception)])}
```

### **3. Cost Optimization (Redis Cache + Batch)**
```python
# Cost: ₹0.35/M tokens → Batch 10 chapters
async def batch_generate_chapters(chapters: list):
    prompt = "\n\n".join([f"Chapter {i}: {chap['title']}" for i, chap in enumerate(chapters)])
    batch_response = await gemini.generate(prompt, max_tokens=32_000)
    
    # Split response → Cache 1hr
    for chap, content in zip(chapters, batch_response.split("\n\n")):
        await redis.setex(f"chapter:{chap['id']}", 3600, content)
```

```python
GEO_CITIES = ["dhemaji", "jorhat", "guwahati", "silchar", "tezpur"]

async def geo_seo_pages(hierarchy, content):
    pages = []
    for city in GEO_CITIES:
        pages.append({
            **hierarchy,
            "page_type": "geo-blog",
            "slug": f"{hierarchy['topic_slug']}-{city}",
            "title": f"{hierarchy['topic_slug']} Notes {city}",
            "content": f"{content}\n\n## For {city} Students"
        })
    return pages
```
