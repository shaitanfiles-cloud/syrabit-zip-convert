## Smart Subject Classifier (chat.py)
```python
SUBJECT_MAP = {
    # Business Studies
    "government role business": "AHSEC Class 12 Business Studies Ch.4",
    "mrp act licensing": "Business Studies Unit 4",
    "privatization": "Business Environment",
    
    # Physics  
    "ohm kirchhoff": "AHSEC Class 12 Physics Ch.3",
    "waves interference": "Physics Ch.8",
    
    # Chemistry
    "p block": "Class 12 Chemistry Ch.7",
    "coordination compounds": "Chemistry Ch.9"
}

def classify_subject(query: str):
    query_lower = query.lower()
    
    # Exact keyword → 95% accuracy
    for keywords, subject in SUBJECT_MAP.items():
        if all(word in query_lower for word in keywords.split()):
            return subject
    
    # Fallback: LLM micro-classify (50 tokens, 100ms)
    subjects = vertexai.classify(query, max_tokens=5)  # "Business Studies"
    return f"AHSEC Class 12 {subjects[0]}"
```

## Dual-Layer Router (Full Flow)

# RAG becomes subject-aware
async def curriculum_rag(query, subject):
    pipeline = [
        {"$match": {
            "status": "published",
            "subject": subject.split()[-1],  # "Business Studies" → "Studies"
            "tags": {"$in": extract_keywords(query)}
        }},
        {"$vectorSearch": {
            "queryVector": embed(query),
            "path": "embedding",
            "numCandidates": 50,  # Scoped = faster
            "limit": 8
        }}
    ]
    return db.seo_pages.aggregate(pipeline)
