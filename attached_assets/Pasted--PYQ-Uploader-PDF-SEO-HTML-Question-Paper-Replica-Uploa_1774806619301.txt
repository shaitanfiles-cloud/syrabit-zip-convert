```python
@app.post("/api/admin/pyq/html-replica")
async def pdf_to_html_replica(file: UploadFile):
    # 1. Gemini Vision OCR (preserve layout)
    ocr_result = await vertex_ocr(file)
    
    # 2. HTML Replica (white BG, black text, exact format)
    html = f"""
<!DOCTYPE html>
<html class="pyq-replica">
<head>
  <title>AHSEC Physics PYQ 2025 - Dhemaji</title>
  <meta name="description" content="AHSEC Class 12 Physics Previous Year Questions 2025">
  <style>
    body {{ background: white; color: black; font-family: 'Times New Roman'; font-size: 14px; margin: 2in 1.5in; line-height: 1.2; }}
    .question {{ margin-top: 20px; font-weight: bold; }}
    .marks {{ color: #666; font-size: 12px; float: right; }}
    .clear {{ clear: both; }}
    @media (max-width: 768px) {{ body {{ margin: 1in; font-size: 12px; }} }}
  </style>
</head>
<body>
  <h1>AHSEC Class 12 Physics PYQ 2025</h1>
"""
    
    for q in ocr_result['questions']:
        html += f"""
  <div class="question">
    <span>Q{q['number']}</span>
    <span class="marks">({q['marks']} marks)</span>
    <div class="clear"></div>
    <p>{q['text']}</p>
  </div>
"""
    
    html += "</body></html>"
    
    # 3. Save HTML + SEO metadata
    seo_doc = {
        "type": "pyq_html_replica",
        "slug": f"ahsec-physics-pyq-{ocr_result['year']}-dhemaji",
        "html_content": html,
        "geo_tags": ["dhemaji", "jorhat", "guwahati", "assam"],
        "schema": {"@type": "ExamPaper"}
    }
    
    await db.seo_pages.insert_one(seo_doc)
    return {"seo_url": f"/pyq/{seo_doc['slug']}"}
```
