## 2. Updated subject_router.py (Syllabus-Driven)
```python
class SubjectRouter:
    async def classify(self, query: str) -> Dict:
        # PRIORITY 1: SYLLABUS MANAGER (98% accuracy)
        syllabus_match = await self._syllabus_db_match(query)
        if syllabus_match:
            return syllabus_match
        
        # PRIORITY 2: Keyword backup
        keyword_match = await self._keyword_match(query)
        if keyword_match:
            return keyword_match
        
        return await self._llm_classify(query)
    
    async def _syllabus_db_match(self, query: str):
        """Syllabus Manager = Source of Truth"""
        vector = await embed_cached(query)
        
        pipeline = [
            {"$vectorSearch": {
                "queryVector": vector,
                "path": "embedding", 
                "index": "syllabus_vector",
                "numCandidates": 20,
                "limit": 3
            }},
            {"$match": {
                "status": "active",
                "similarity": {"$gte": 0.75}  # Strict threshold
            }},
            {"$sort": {"similarity": -1}},
            {"$limit": 1}
        ]
        
        doc = await self.db.syllabus.aggregate(pipeline).next()
        if doc:
            return {
                "board": doc['board'],
                "class": doc['class_level'],
                "stream": doc['stream'], 
                "subject": doc['subject'],
                "chapter": doc['chapter_title'],
                "unit": doc.get('unit', ''),
                "confidence": doc['similarity'],
                "syllabus_id": doc['_id']
            }
        return None
```
