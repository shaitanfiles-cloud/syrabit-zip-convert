## **Code Wiring (15min Deploy)**
```jsx
// LessonCard.jsx:892
const handleAiGenerate = async () => {
  setGenerating(true)
  const notes = await fetch(`/api/admin/lessons/${chapterId}/generate-notes`, {
    method: 'POST',
    body: JSON.stringify({chapter_name, syllabus_content})
  })
  setWordCount(notes.wordCount)
  setPreview(notes.preview)
  // Auto-trigger Agentic if toggled
  if (autoAgentic) triggerAgentic(chapterId)
}

// server.py
@app.post("/api/admin/lessons/{chapter_id}/generate-notes")
async def generate_lesson_notes(chapter_id: str, chapter_name: str, syllabus_content: str):
    prompt = f"Chapter: {chapter_name}\nSyllabus: {syllabus_content}\nGenerate detailed notes..."
    notes = await gemini.generate(prompt)
    await db.chapters.update_one({"_id": chapter_id}, {"$set": {"notes": notes}})
    return {"wordCount": len(notes.split()), "preview": notes[:130]}
```
