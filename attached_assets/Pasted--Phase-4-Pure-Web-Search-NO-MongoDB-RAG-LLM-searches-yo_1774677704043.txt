## Phase 4: Library Web Search (5 steps, NO Mongo Vectors)

### 1. **Syrabit Search Tool** (`backend/tools/syrabit_web_search.py`)
```python
# Pure web search: ONLY syrabit.ai/library pages
def syrabit_web_search(query: str, board: str = None, class_num: str = None):
    base_query = f'site:syrabit.ai/library "{query}"'
    if board: base_query += f' "{board}"'
    if class_num: base_query += f' "class {class_num}"'
    
    url = f"https://www.google.com/search?q={urllib.parse.quote(base_query)}"
    headers = {"User-Agent": "Mozilla/5.0 (compatible; SyrabitBot/1.0)"}
    
    resp = requests.get(url, headers=headers)
    soup = BeautifulSoup(resp.text, 'html.parser')
    
    results = []
    for g in soup.find_all('div', class_='g')[:5]:
        title = g.find('h3')
        link = g.find('a', href=True)
        snippet = g.find('span', class_='st')
        if title and link:
            results.append({
                "title": title.get_text(),
                "url": link['href'],
                "snippet": snippet.get_text()[:200] if snippet else ""
            })
    return results
```

### 2. **Chat ALWAYS Uses Web Search** (`backend/chat.py`)
```python
@app.route("/chat", methods=["POST"])
def chat():
    data = request.json
    question = data["question"]
    
    # ALWAYS web search syrabit.ai (NO Mongo RAG)
    sources = syrabit_web_search(question, data.get("board"), data.get("class"))
    
    context = "\n".join([
        f"[{i+1}] {s['title']}\n{s['url']}\n{s['snippet']}\n---"
        for i, s in enumerate(sources)
    ])
    
    prompt = f"""AHSEC/SEBA Tutor. Answer ONLY from syrabit.ai sources below.

SOURCES FROM SYRABIT LIBRARY:
{context}

Q: {question}

Cite with [#] after relevant parts. ALWAYS include sources."""

    response = llm.generate(prompt)
    return {
        "answer": response,
        "sources": sources  # ALWAYS clickable
    }
```
