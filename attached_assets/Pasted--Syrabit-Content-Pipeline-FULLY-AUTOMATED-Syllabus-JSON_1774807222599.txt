## **FastAPI Automation Pipeline**
```python
@app.post("/api/admin/pipeline/auto-generate")
async def auto_content_pipeline(subject_id: str):
    """1-click: Syllabus → 100 SEO assets"""
    
    # 1. Get syllabus context
    subject = await db.subjects.find_one({"_id": ObjectId(subject_id)})
    
    assets = []
    
    # 2. Generate chapters (12 chapters)
    for chapter in subject['chapters']:
        # Chapter content
        content = await gemini_generate_chapter(chapter['title'], subject)
        
        # PYQ replica HTML
        pyq_html = await pdf_to_html_replica(chapter['pyq_pdf'])
        
        # MCQs + Flashcards
        mcqs = await generate_mcqs(content)
        flashcards = await generate_flashcards(content)
        
        # SEO Blog
        blog = await generate_seo_blog(subject, chapter, content, mcqs)
        
        assets.extend([content, pyq_html, mcqs, flashcards, blog])
    
    # 3. Bulk publish
    await db.seo_pages.insert_many(assets)
    await ping_sitemap_google()
    
    return {
        "generated": len(assets),
        "blogs": 12,
        "pyqs": 12,
        "mcqs": 300,
        "seo_urls": [a['slug'] for a in assets if 'blog' in a['type']]
    }
```

## **Geo + SEO Auto-Optimization**
```python
GEO_LOCATIONS = ["dhemaji", "jorhat", "guwahati", "silchar", "tezpur"]

def generate_seo_blog(subject, chapter, content):
    for geo in GEO_LOCATIONS:
        return {
            "title": f"{subject.board} {subject.class} {subject.name} Ch.{chapter.num} {geo}",
            "slug": f"{subject.board.lower()}-{subject.class}-{chapter.slug}-{geo}",
            "content": content,
            "meta": {
                "description": f"{subject.name} notes Ch.{chapter.num} for {geo} students",
                "keywords": [f"{subject.name} {geo}", f"pyq {geo}"]
            }
        }
```
