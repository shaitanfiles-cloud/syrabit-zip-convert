Fix 2: Bulk Skip (5min)

@app.post("/subjects/{id}/generate-notes-bulk")
async def bulk_notes(subject_id: str, skip_existing: bool = False):
    chapters = await db.chapters.find({"subject_id": subject_id}).to_list(None)
    for c in chapters:
        if skip_existing and c.get('notes', '').strip().count(' ') > 50: continue
        await generate_notes(c['_id'])

@app.get("/content/library-bundle")
async def library_bundle():
    subjects = await db.subjects.aggregate([
        {"$lookup": {"from": "chapters", "localField": "_id", "foreignField": "subject_id", "as": "chapters"}},
        {"$addFields": {
            "notes_pct": {"$divide": [{"$size": "$chapters"}, {"$size": {"$objectToArray": "$chapters"}}]},
            "pyq_count": {"$sum": "$chapters.pyqs.length"},
            "flash_count": {"$sum": "$chapters.flashcards.length"}
        }}
    ]).to_list(None)
