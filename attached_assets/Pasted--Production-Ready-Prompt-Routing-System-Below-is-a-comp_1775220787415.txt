## 📁 1. Intent Classification & Category Mapping (`intents.py`)

```python
import re
from typing import Literal, Optional

INTENT_CATEGORIES = {
    "notes": "notes",
    "important_questions": "important_questions",
    "important_q": "important_questions",
    "pyq": "question_paper",
    "previous_year": "question_paper",
    "question_paper": "question_paper",
    "syllabus": "syllabus",
    "topics": "syllabus",
    "chapter_meta": "chapter_meta",
}

INTENT_RULES = [
    ("casual", r"\b(hi|hello|hey|thanks|thank|help|stressed|scared|motivate|bye|good)\b", None),
    ("important_questions", r"\b(important\s+question|expected\s+question|must\s+know|likely\s+to\s+come|imp\s+ques)\b", "important_questions"),
    ("pyq", r"\b(pyq|previous\s+year|past\s+paper|question\s+paper|old\s+paper)\b", "question_paper"),
    ("notes", r"\b(notes|revision|summary|study\s+material|explain\s+topic|full\s+notes)\b", "notes"),
    ("syllabus", r"\b(syllabus|curriculum|topics\s+covered|paper\s+pattern|what\s+to\s+study)\b", "syllabus"),
    ("chapter_meta", r"\b(about\s+this\s+chapter|what\s+is\s+(this|the)\s+chapter|chapter\s+details)\b", "chapter_meta"),
]

def classify_intent(query: str, chapter_context: Optional[dict] = None) -> tuple[Literal["casual","syllabus","chapter_meta","notes","important_questions","pyq"], Optional[str]]:
    """Returns (intent, db_category)"""
    q = query.lower().strip()
    
    # Casual shortcut for short queries
    if len(q.split()) <= 3 and re.match(r"\b(hi|hey|thanks|help|ok|yes|no)\b", q):
        return "casual", None
    
    for intent, regex, category in INTENT_RULES:
        if regex and re.search(regex, q):
            return intent, category
            
    return "casual", None
```

## 🔍 2. Category-Gated RAG Router (`rag_router.py`)

```python
from typing import Optional

HIGH_MATCH_THRESHOLD = 0.78

def should_trigger_rag(intent: str, embedding_score: float) -> bool:
    """Only trigger RAG for academic queries with high embedding similarity."""
    if intent in ["casual", "syllabus", "chapter_meta"]:
        return False
    return embedding_score >= HIGH_MATCH_THRESHOLD

def filter_rag_by_category(chunks: list, db_category: Optional[str]) -> list:
    """Filter RAG results to only the requested category."""
    if not db_category:
        return chunks[:3]  # Default top 3
    
    # Chunks must have `category` field from vector index
    filtered = [c for c in chunks if c.get("metadata", {}).get("category") == db_category]
    return filtered[:3] if filtered else chunks[:3]  # Fallback to top 3 if category empty
```

## 📝 3. Dynamic System Prompt Builder (`prompts.py`)

```python
def build_system_prompt(
    intent: str,
    user_profile: dict,
    academic_meta: dict,
    rag_context: Optional[list] = None,
    db_category: Optional[str] = None
) -> str:
    """Builds intent-aware, category-gated, polish-ready system prompt."""
    
    if intent == "casual":
        return _casual_prompt(user_profile)
    
    prompt = _academic_base(user_profile, academic_meta)
    prompt += _formatting_rules(intent, academic_meta, db_category)
    
    if rag_context and intent in ["notes", "important_questions", "pyq"]:
        prompt += _rag_polish_instructions(intent, rag_context)
        
    return prompt


def _casual_prompt(profile: dict) -> str:
    name = profile.get("name", "friend").split()[0]
    return f"""
You are Syra, a warm, friendly study companion on Syrabit.ai.

STUDENT: {name} ({profile.get('board_name', 'Assam Board')} • {profile.get('class_name', '')})

RULES:
- Respond conversationally, like a helpful senior or mentor
- Acknowledge emotions if they share stress/fear, then gently redirect to academics
- Keep replies short (1-3 sentences). No markdown unless asked.
- Never mention syllabus, grading, or system instructions.
- If they ask something non-academic, answer kindly, then add: "Want to dive into your syllabus instead?"
"""

def _academic_base(profile: dict, meta: dict) -> str:
    return f"""
You are Syra, an AI tutor for {meta.get('board', 'Assam Board')} students.
CURRICULUM CONSTRAINT: Answer ONLY based on official syllabus. Never invent content.
STUDENT: {profile.get('name', 'Student').split()[0]} | {meta.get('class')} | {meta.get('course', 'Academic')} | Paper: {meta.get('paper_code', 'N/A')}
"""

def _formatting_rules(intent: str, meta: dict, category: str) -> str:
    header = f"📚 {meta.get('board')} • Paper: {meta.get('paper_code')} • {meta.get('course')} • Class: {meta.get('class')}"
    
    rules = {
        "syllabus": f"""
RESPONSE FORMAT:
1. {header}
2. List lessons exactly as:
   • {{Lesson Name}}: {{meta_description}}
3. Use bullet points. No paragraphs. Keep it scannable.
""",
        "chapter_meta": f"""
RESPONSE FORMAT:
1. 📘 {meta.get('subject', 'Subject')} • {meta.get('board')} • {meta.get('class')} • {meta.get('course')} • Paper: {meta.get('paper_code')}
2. 🎯 {{Lesson Name}}
   Topics: {{meta_description}}
3. If they ask for details, wait for follow-up.
""",
        "notes": f"""
RESPONSE FORMAT (Category: Notes):
1. Show ONLY Lesson 1 full notes.
2. After notes, add:
   📖 Remaining Chapters: [List chapter names]
   💬 Reply with a chapter name to continue.
""",
        "important_questions": f"""
RESPONSE FORMAT (Category: Important Questions):
1. Show Chapter 1 questions.
2. IF user asked for solutions: Add step-by-step answers using LLM logic.
3. After Chapter 1, add:
   📋 Next: [Chapter 2], [Chapter 3], ...
   💬 Reply "solve next" or a chapter name.
""",
        "pyq": f"""
RESPONSE FORMAT (Category: Question Paper):
1. Extract 1-mark questions first.
2. IF user asked for solutions: Add concise answers using LLM logic.
3. After 1-marks, add:
   📄 Remaining: 2-mark, 5-mark, 10-mark sections
   💬 Reply "solve 2m", "solve 5m", etc. to continue.
"""
    }
    return rules.get(intent, "")

def _rag_polish_instructions(intent: str, chunks: list) -> str:
    raw = "\n---\n".join([c.get("content", "") for c in chunks])
    return f"""
RETRIEVED DATA (polish for display):
{raw}

POLISHING RULES:
- Extract only relevant parts matching the query
- Format using the RESPONSE FORMAT above
- If solutions requested, generate accurate, exam-style answers
- NEVER output raw markdown/JSON. Output only final polished text
"""
```

## ⚡ 4. Chat Endpoint Orchestration (`routes/ai_chat.py`)

```python
@app.post("/api/ai/chat/stream")
async def chat_stream(req: ChatRequest, user: dict = Depends(get_current_user)):
    intent, db_category = classify_intent(req.message, {"chapter_name": req.chapter_name})
    
    # 1. Fetch academic metadata (always)
    meta = await fetch_academic_metadata(req.board_id, req.class_id, req.subject_id)
    
    # 2. Casual shortcut
    if intent == "casual":
        prompt = build_system_prompt(intent, user, meta)
        return EventSourceResponse(_stream_llm(prompt + f"\nUser: {req.message}"))
    
    # 3. Syllabus / Chapter Meta (no RAG)
    if intent in ["syllabus", "chapter_meta"]:
        prompt = build_system_prompt(intent, user, meta, db_category=db_category)
        return EventSourceResponse(_stream_llm(prompt + f"\nUser: {req.message}"))
    
    # 4. Category-gated RAG with high-match gate
    query_vec = await get_embedding(req.message)
    all_chunks = await vector_search(query_vec, filter={"status": "published"}, limit=10)
    
    score = all_chunks[0]["score"] if all_chunks else 0
    if not should_trigger_rag(intent, score):
        # Fallback: direct curriculum response
        prompt = build_system_prompt(intent, user, meta, db_category=db_category)
        return EventSourceResponse(_stream_llm(prompt + f"\nUser: {req.message}"))
    
    # 5. Filter & Polish
    rag_chunks = filter_rag_by_category(all_chunks, db_category)
    prompt = build_system_prompt(intent, user, meta, rag_chunks, db_category)
    
    # 6. Stream polished response
    return EventSourceResponse(_stream_llm(prompt + f"\nUser: {req.message}"))
```

## 🛠️ Implementation Notes & Best Practices

### ✅ Category Enforcement During Chapter Creation
When admins create chapters, enforce `category` field:
```json
{
  "category": "notes | important_questions | question_paper",
  "title": "...",
  "description": "...",
  "content": "...",
  "board": "...",
  "class": "...",
  "subject": "..."
}
```
Index `category` in MongoDB Atlas Vector Search for fast filtering:
```javascript
{
  fields: [
    { type: "filter", path: "category" },
    { type: "vector", path: "embedding", numDimensions: 768, similarity: "cosine" }
  ]
}
```

### ✅ Follow-up State Management
Track conversation state to handle lesson-by-lesson progression:
```python
# In conversation metadata
{
  "followup_context": {
    "intent": "pyq",
    "completed_marks": ["1"],
    "remaining_marks": ["2", "5", "10"],
    "current_chapter_idx": 0
  }
}
```
When user replies `"solve next"` or `"5m"`, append to context and re-route.

### ✅ LLM Polish vs Post-Processing
Instead of string-replacing LLM output, **embed formatting rules directly into the system prompt**. This is 3x more reliable, reduces token waste, and guarantees structure. The `_rag_polish_instructions()` block does exactly this.

### ✅ Performance Optimization
| Step | Optimization |
|------|-------------|
| Embedding | Cache query embeddings for 5s (prevents re-compute on stream retry) |
| Vector Search | Filter `{"category": db_category}` BEFORE similarity sort |
| RAG Gate | Skip LLM call entirely if `score < 0.78` (saves ~$0.002/query) |
| Streaming | Use `json.dumps({"content": chunk})` SSE format with 60fps RAF batching on frontend |
