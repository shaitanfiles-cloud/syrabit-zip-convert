    if not is_nep_pdf(await file.read()):

    for course in courses:
        course_type = detect_nep_course(course['code'], course['title'])
        
        # 3. DEGREE Hierarchy Only
        board = await get_or_create_board("DEGREE-" + course['college'])
        semester = await get_or_create_class(course['semester'])
        stream = await get_or_create_stream(course_type)
        subject = await create_subject(course['title'], course_type)
        
        # 4. Embed Chapters

            await embed_chapter(chapter['_id'], unit['content'])
            created_count += 1
