```python
@app.post("/api/admin/chapters/{subject_id}")
async def create_chapter(
    subject_id: str,
    chapter_data: ChapterUpdate,
    admin: dict = Depends(get_admin_user)
):
    # 1. Fetch subject context
    subject = await subjects_collection.find_one({"_id": subject_id})
    if not subject: raise HTTPException(404, "Subject not found")
    
    # 2. Extract topics from description (fallback to manual if provided)
    topics = extract_topics_from_description(chapter_data.description)
    
    # 3. Create chapter document
    chapter_doc = {
        "subject_id": subject_id,
        "board_id": subject["board_id"],
        "class_id": subject["class_id"],
        "stream_id": subject.get("stream_id"),
        "title": chapter_data.title,
        "slug": chapter_data.slug,
        "description": chapter_data.description,
        "topics": topics,
        "content": None,
        "content_type": "notes",
        "status": "draft",
        "seo_meta": {},
        "created_at": datetime.utcnow(),
        "updated_at": datetime.utcnow()
    }
    
    result = await chapters_collection.insert_one(chapter_doc)
    chapter_id = str(result.inserted_id)
    
    # 4. Generate embeddings for each topic + chapter description
    await generate_and_store_embeddings(
        chapter_id=chapter_id,
        texts=[chapter_data.description] + topics,
        metadata={
            "type": "chapter_topic",
            "board": subject["board_id"],
            "class": subject["class_id"],
            "stream": subject.get("stream_id"),
            "subject": subject_id
        }
    )
    
    return {"chapter_id": chapter_id, "status": "created", "topics_indexed": len(topics)}
```

```python
@app.post("/api/admin/chapters/{chapter_id}/generate-notes")
async def generate_chapter_notes(

):
    chapter = await chapters_collection.find_one({"_id": ObjectId(chapter_id)})

    prompt = f"""
Generate high-quality, curriculum-aligned study notes for the chapter "{chapter['title']}".
STRICT FOCUS: Cover ONLY these topics from the chapter description:
{', '.join(chapter['topics'])}

REQUIREMENTS:
- Use clear H2/H3 headings
- Include definitions, key points, and 1-2 exam-style examples per topic
- Match {chapter['board_id']} marking scheme language
- Keep it concise (800-1200 words)
- NO intros/outros, start directly with content
"""
    
    # Generate via your LLM pool (use fastest reliable model)
    notes = await generate_llm_response(prompt, model="groq/llama-3.3-70b")
    
    # Save & update
    await chapters_collection.update_one(
        {"_id": ObjectId(chapter_id)},
        {"$set": {
            "content": notes,
            "status": "generated",
            "updated_at": datetime.utcnow()
        }}
    )
    
    return {"status": "success", "word_count": len(notes.split())}
```

## 🌐 2. SEO Meta Generation + Direct Publish

### A. AI-Generated SEO Meta via Web Search (`seo_engine.py`)
```python
@app.post("/api/admin/chapters/{chapter_id}/publish")
async def publish_chapter(
    chapter_id: str,
    admin: dict = Depends(get_admin_user)
):
    chapter = await chapters_collection.find_one({"_id": ObjectId(chapter_id)})
    if not chapter or not chapter["content"]:
        raise HTTPException(400, "Chapter content missing")
    
    # 1. Fetch high-search queries via Tavily/DuckDuckGo
    search_query = f"{chapter['board_id']} {chapter['subject_id']} {chapter['title']} notes"
    web_results = await fetch_top_search_results(search_query, limit=5)
    
    # 2. Generate SEO meta
    seo_prompt = f"""
Based on these top search results and chapter content, generate SEO meta:
{json.dumps(web_results)}

CHAPTER: {chapter['title']}
BOARD: {chapter['board_id']} | CLASS: {chapter['class_id']}

Return ONLY JSON:
{{"title": "50-60 chars", "description": "120-160 chars", "keywords": ["kw1","kw2","kw3"]}}
"""
    seo_meta = await generate_llm_response(seo_prompt, response_format="json")
    
    # 3. Validate lengths
    if len(seo_meta["title"]) < 50 or len(seo_meta["title"]) > 60:
        raise HTTPException(400, "SEO title must be 50-60 chars")
    if len(seo_meta["description"]) < 120 or len(seo_meta["description"]) > 160:
        raise HTTPException(400, "SEO description must be 120-160 chars")
    
    # 4. Publish
    await chapters_collection.update_one(
        {"_id": ObjectId(chapter_id)},
        {"$set": {
            "seo_meta": seo_meta,
            "status": "published",
            "published_at": datetime.utcnow()
        }}
    )
    
    # 5. Update subject content card counts
    await subjects_collection.update_one(
        {"_id": chapter["subject_id"]},
        {"$inc": {"notes_count": 1}, "$set": {"updated_at": datetime.utcnow()}}
    )
    
    return {"status": "published", "seo": seo_meta}
```

## ⚡ 3. RAG Overhaul: Fast, High-Match, Intent-Aware

### A. High-Match Trigger Logic (`rag.py`)
```python
async def retrieve_context(query: str, user_context: dict, intent: str) -> list:
    """Fast RAG with high-match gating + intent preprocessing."""
    # 1. Embed query
    query_vec = await get_embedding(query)
    
    # 2. Vector search against chapter topic embeddings
    results = await vector_search(
        collection="chapters",
        query_vector=query_vec,
        filter={"status": "published"},
        limit=10
    )
    
    # 3. HIGH-MATCH GATE
    if not results or results[0]["score"] < 0.78:
        return []  # Skip RAG, fall back to direct LLM or syllabus guard
    
    # 4. Intent-aware preprocessing
    filtered_context = preprocess_for_intent(results, intent, user_context)
    
    return filtered_context

def preprocess_for_intent(results: list, intent: str, user: dict) -> list:
    """Filter & format RAG data per user needs before LLM."""
    # Board/class filtering
    user_board = user.get("board_id")
    filtered = [r for r in results if r["board"] == user_board]
    if not filtered: filtered = results  # fallback to all
    
    # Intent formatting
    context_blocks = []
    for r in filtered[:3]:  # Top 3 only
        block = f"[Source: {r['subject_id']} | Chapter: {r['title']}]\n{r['content_preview']}"
        if intent == "formula":
            block += "\n→ Extract only formulas, define variables, show 1 example"
        elif intent == "important_questions":
            block += "\n→ List only exam-weighted Q&A, tag marks/year"
        context_blocks.append(block)
    
    return context_blocks
```

### B. Caching & Speed
- **Redis cache key:** `rag:{query_hash}:{user_board}` (TTL: 24h)
- **In-memory LRU cache** for recent queries
- **Embedding cache:** Store query embeddings for 5s to prevent re-computation on stream retry

### B. SERP-Aware Share Hook (`src/hooks/useShare.jsx`)
```jsx
export function useShare({ title, description, url }) {
  const handleShare = async () => {
    // Mobile: Native share
    if (navigator.share) {
      await navigator.share({ title, text: description, url });
      return;
    }
    // Desktop: Copy + toast
    await navigator.clipboard.writeText(url);
    toast.success("Link copied! SERP preview will show on WhatsApp/Twitter.");
  };
  return { handleShare };
}
```
