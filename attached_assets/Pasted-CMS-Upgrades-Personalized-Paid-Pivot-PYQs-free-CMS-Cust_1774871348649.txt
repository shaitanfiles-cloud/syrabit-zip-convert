Fix 1: User CMS Routes (10min)
python
@app.route("/cms/{user_id}/{slug}", methods=["GET"])
async def user_cms(user_id: str, slug: str, user=Depends(get_user)):
    if user.id != user_id or not user.is_paid():
        raise HTTPException(403, "Private Study Plan")
    
    doc = await db.cms_documents.find_one({
        "user_id": user_id,
        "slug": slug,
        "type": "personalized"
    })
    return doc
Fix 2: Razorpay Subscribe (20min)
python
@app.post("/subscribe")
async def razorpay_checkout(user=Depends(get_user)):
    order = razorpay.order.create({
        "amount": 1000,  # ₹10
        "currency": "INR",
        "notes": {"user_id": user.id}
    })
    return {"id": order.id, "amount": order.amount}
Fix 3: Auto-Plans (25min)
python
@app.post("/cms/personalize")
async def generate_plan(user=Depends(get_user)):
    weak_topics = await get_user_weaknesses(user.id)
    plan = await gemini.generate(f"Your Physics gaps: {weak_topics}\n7-day plan:")
    doc_id = await db.cms_documents.insert_one({
        "user_id": user.id,
        "type": "personalized",
        "slug": f"physics-plan-{datetime.now().isoformat()}",
        "content": plan
    })
    return {"url": f"/cms/{user.id}/{doc_id}"}
