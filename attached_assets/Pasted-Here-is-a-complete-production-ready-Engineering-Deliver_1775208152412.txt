## 🔐 2. Security Headers Middleware (FastAPI + Env Toggles)

**File:** `app/middleware/security.py`
```python
import os
from fastapi import Request, Response
from starlette.middleware.base import BaseHTTPMiddleware

class SecurityHeadersMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next) -> Response:
        # Skip for OPTIONS/preflight to avoid CORS conflicts
        if request.method == "OPTIONS":
            return await call_next(request)
            
        response = await call_next(request)
        
        toggles = {
            "hsts": os.getenv("SEC_HSTS", "true").lower() == "true",
            "xcto": os.getenv("SEC_XCTO", "true").lower() == "true",
            "xframe": os.getenv("SEC_XFRAME", "true").lower() == "true",
            "referrer": os.getenv("SEC_REFERRER", "true").lower() == "true",
            "permissions": os.getenv("SEC_PERM", "true").lower() == "true",
            "csp_report": os.getenv("SEC_CSP_REPORT_ONLY", "false").lower() == "true",
        }

        if toggles["hsts"]:
            response.headers["Strict-Transport-Security"] = "max-age=31536000; includeSubDomains; preload"
        if toggles["xcto"]:
            response.headers["X-Content-Type-Options"] = "nosniff"
        if toggles["xframe"]:
            response.headers["X-Frame-Options"] = "DENY"
        if toggles["referrer"]:
            response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"
        if toggles["permissions"]:
            response.headers["Permissions-Policy"] = "geolocation=(), microphone=(), camera=(), payment=()"
        if toggles["csp_report"]:
            response.headers["Content-Security-Policy-Report-Only"] = (
                "default-src 'self'; "
                "script-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net https://unpkg.com; "
                "style-src 'self' 'unsafe-inline'; "
                "img-src 'self' https: data:; "
                "connect-src 'self' https://*.syrabit.ai https://api.groq.com https://api.openrouter.ai https://generativelanguage.googleapis.com; "
                "report-uri https://syrabit.ai/api/security/csp-report;"
            )
        return response
```

## 🛡️ 4. Prompt Safety & Edge Case Guardrails Module

**File:** `app/guardrails/prompt_safety.py`
```python
import re
from typing import Optional

OUT_OF_SCOPE = "I focus on AHSEC syllabus support. I can't assist with that, but I'm here for your coursework."
ACADEMIC_INTEGRITY = "I won't provide direct exam answers, but I can walk you through the concept step-by-step using your syllabus."
SENSITIVE_FALLBACK = "I'm designed for academic support. If you're struggling with personal or mental health topics, please reach out to a trusted adult or helpline."
INJECTION_PATTERNS = re.compile(r"(?i)\b(ignore|override|disregard|pretend|act as|system prompt|jailbreak)\b")
ACADEMIC_CHEAT = re.compile(r"(?i)\b(exam|test|cheat|leak|answer key|paper solution|direct answer)\b")
SENSITIVE_KEYWORDS = re.compile(r"(?i)\b(suicide|self.?harm|abuse|violence|explicit|drugs|bomb|weapon)\b")

def evaluate_prompt_safety(prompt: str) -> tuple[str, Optional[str]]:
    """
    Returns: (safe_prompt or fallback_response, detected_intent_modifier)
    """
    if INJECTION_PATTERNS.search(prompt):
        return OUT_OF_SCOPE, "injection_blocked"
    if ACADEMIC_CHEAT.search(prompt) and re.search(r"(?i)(give|send|show|provide)\s+(answer|solution|key)", prompt):
        return ACADEMIC_INTEGRITY, "academic_integrity"
    if SENSITIVE_KEYWORDS.search(prompt):
        return SENSITIVE_FALLBACK, "sensitive_content"
    return prompt, None

def validate_llm_output(chunk: str) -> bool:
    """Post-generation check for streaming safety. Returns False if policy violation detected."""
    return not re.search(r"(?i)(here's how to cheat|ignore rules|bypass|unsafe|exploit)", chunk)
```
