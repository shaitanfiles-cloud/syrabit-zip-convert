```python
from pymongo import MongoClient

@app.post("/rag/search")
async def rag_search(query: str):
    # Vector search ALL published pages
    pipeline = [
        {"$match": {"status": "published"}},
        {"$search": {
            "index": "syrabit_vector",  # Atlas Search
            "knnBeta": {"vector": embed(query), "path": "embedding", "k": 12}
        }},
        {"$project": {"content": 1, "slug": 1, "chunk": {"$substr": ["$content", 0, 800]}}}
    ]
    results = db.seo_pages.aggregate(pipeline)
    return list(results)  # 12 chunks to Gemini
```

## 4. **Full Chat Endpoint**
```python
@app.post("/chat")
async def syrabit_chat(request: ChatRequest):
    chunks = await rag_search(request.message)  # Your 10 published pages
    grounding = "\n".join([f"[PAGE:{c['slug']}]\n{c['chunk']}" for c in chunks])
    
    response = model.generate_content(
        PROMPT.format(rag_content=grounding, user_message=request.message),
        generation_config={"temperature": 0.05}
    )
    return {
        "reply": response.text,
        "sources": [c['slug'] for c in chunks]
    }
```
